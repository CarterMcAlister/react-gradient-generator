## Install

```bash
npm install --save react-gradient-generator
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-gradient-generator'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
